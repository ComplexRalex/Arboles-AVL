878  // Si el vertice actual es NULL (no hay camino actual->destino), se borra todo el
879  // progreso en la lista
880  if(actual == NULL){
881  while(!(*camino)->vacia())
882  (*camino)->suprimir_final();

887  // termina la ejecucion (ingresando el ultimo nodo al camino)
888  if(actual == destino){
889  (*camino)->insertar_final(actual);
890  return;
891  }
892

894  Vertice** hijos = obtener_hijos(actual);
895

897  (*camino)->insertar_final(actual);

899  if(destino->dato < actual->dato)
900  __camino__(hijos[0],destino,camino);
901  else
902  __camino__(hijos[1],destino,camino);
