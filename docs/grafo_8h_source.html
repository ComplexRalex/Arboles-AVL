373  map<string, string> pertenece; // diccionario que indica a que arbol pertenece el vertice
374  //nuevoGrafo->inicializar(); ??
375  int numVertices = cardinalidad(), numAristas;
376  Lista<Arista*> comparando = aristas;
377 
378  if(vertices.tamanio() != 0 && aristas.tamanio() != 0){
379  for(Lista<Vertice*>::iterator i = vertices.begin(); i != vertices.end(); i++){
380  // el diccionario 'pertenece' almacena en cada posicion
381  // el arbol al que pertenece el vertice en dicha posicion
382  // con un ejemplo de los siguientes vertices: [A, B, C, D]
383  // se inicializa como: pertenece = [A:A, B:B, C:C, D:D] indicando que
384  // el vertice en la posicion A pertenece al arbol A
385  // el vertice en la posicion D pertenece al arbol D
386  // posteriomente se utiliza para detectar los posibles ciclos,
387  // ya que si los extremos de una arista pertenecen a un mismo
388  // arbol se formara un ciclo
389  pertenece[(*i)->nombre] = (*i)->nombre;
390  }
391  numAristas = 0;
392  while(numAristas < numVertices - 1 && !comparando.vacia()){
393  Arista* minimo = comparando.primero();
394 
395  for(Lista<Arista*>::iterator i = comparando.begin(); i != comparando.end(); i++)
396  if((*i)->peso < minimo->peso)
397  minimo = *i;
398 
399  // minimo es una arista: (Vertice* U, Vertice* V, float Peso)
400  string U = minimo->origen->nombre,
401  V = minimo->destino->nombre;
402 
403  if(pertenece[U] != pertenece[V]){
404  nuevoGrafo->insertar_vertice(U);
405  nuevoGrafo->insertar_vertice(V);
406  nuevoGrafo->insertar_arista(nuevoGrafo->get_vertice(U),nuevoGrafo->get_vertice(V),minimo->peso);
407  nuevoGrafo->insertar_arista(nuevoGrafo->get_vertice(V),nuevoGrafo->get_vertice(U),minimo->peso);
408  numAristas ++;
409  // todos los vertices que estaban en el arbol V ahora

411  string fcc = pertenece[V];
412  pertenece[V] = U;
413  for(map<string, string>::value_type & elem : pertenece){
414  if(elem.first == fcc)

416  }
